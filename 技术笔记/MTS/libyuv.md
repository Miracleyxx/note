```
cmake -DCMAKE_INSTALL_PREFIX=/home/lyc/libyuv/libyuv_lib \
      -DCMAKE_BUILD_TYPE="Release" \
      -DUNIT_TEST=OFF \
      -DBUILD_SHARED_LIBS=OFF \
```
